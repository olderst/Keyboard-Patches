# Supported models: Acer Aspire V3-771g

# works with:
#  ELAN
#  Synaptic

into method label _Q71 replace_content
begin
// Brightness Down\n
Notify(\RMKB, 0x216b)\n
Notify(\RMKB, 0x226b)\n
end;

into method label _Q70 replace_content
begin
// Brightness Up\n
Notify(\RMKB, 0x2171)\n
Notify(\RMKB, 0x2271)\n
end;
